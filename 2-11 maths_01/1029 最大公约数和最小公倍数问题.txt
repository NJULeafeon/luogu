#include <iostream>
#include <cmath>
#include <cstring>
#include <vector>
using namespace std;
int gcd(int i,int j){
    if(i < j) {
        int temp = j;
        j = i;
        i = temp;
    }
    if(j == 0) return i;
    gcd(j,i%j);
}
int main() {
    freopen("in.txt","r",stdin);
    int ans = 0;
    int a,b;
    cin >> a >> b;
    int aa[100002];
    memset(aa,0, sizeof(aa));
    for (int i = a; i <= b; i+=a) {
        aa[i]++;
    }
    for (int j = 1; j <= sqrt(b); ++j) {
        if(b % j == 0){
            if(j == sqrt(b)) aa[j]++;
            else {aa[j]++;aa[b/j]++;}
        }
    }
    vector <int>list;
    int cnt = 0;
    for (int k = a; k <= b; ++k) {
        if(aa[k] == 2) list.push_back(k);
    }
    for (int i = 0; i < list.size(); ++i) {
        for (int j = i; j < list.size(); ++j) {
            int gcdd = gcd(list[i],list[j]);
            if(gcdd != a) continue;
            if(list[i]*list[j]/gcdd != b)continue;
            ans+=2;
        }
    }
    cout << ans;
    return 0;
}
