//RE 习惯性把sort(a,a+m,cmp)打成(a,a+n,cmp)
#include <iostream>
#include <algorithm>

using namespace std;
typedef struct _milk{
    int price;
    int amount;
}milk;
bool cmp(milk m1,milk m2){
    return m1.price < m2.price;
}
int main() {
    freopen("in.txt","r",stdin);
    int n,m;
    cin >> n >> m;
    milk a[5002];
    for (int i = 0; i < m; ++i) {
        cin >> a[i].price >> a[i].amount;
    }
    sort(a,a+m,cmp);
    int ans = 0;
    int num = 0;
    int cnt = 0;
    while(num < n)
    {
        if(a[cnt].amount <= n - num)
        {
            ans += a[cnt].price * a[cnt].amount;
            num += a[cnt].amount;
            cnt++;
        }
        else {
            ans += a[cnt].price * (n - num);
            break;
        }
    }
    cout << ans;
    return 0;
}
